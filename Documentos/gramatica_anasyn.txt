#PSEUDO-GRAMATICA (NO FORMALISMO DE BACK-NAUR) PARA O ANALISADOR SINTÁTICO
<instrução>     ::= <expressão>
<expressão>     ::= ("!") <numero> ("+"|"-"|"*"|"/" <numero>)
<numero>        ::= <milhar>
<milhar>        ::= <milhar> | <centena> "e" <milhar>
<centena>       ::= <centena> | <centena> "e" <dezena>
<dezena>        ::= <dezena> | <dezena> "e" <unidade>
<unidade>       ::= <alfabeto> <mais_alfabeto>
<alfabeto>      ::= "a" | "b" | "c" | "d" | "e" | "f" | "g" | "h" | "i" | "j" | "k" | "l" | "m" | "n" | "o" | "p" | "q" | "r" | "s" | "t" | "u" | "v" | "w" | "y" | "x" | "z"
